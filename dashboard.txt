    function shortUrl() {
      const url = document.getElementById('longurl').value;
      if(!url) return;
      const res = document.getElementById('shorturl');

    let pomoSec = 25*60;
    let pomoInterval;
    function setPomo(min) {
      clearInterval(pomoInterval);
      pomoSec = min*60;
      document.getElementById('pomodisplay').innerText = min+":00";
      document.getElementById('p25').className = min===25 ? "flex-1 p-2 rounded active-pomo" : "flex-1 p-2 rounded inactive-pomo";
      document.getElementById('p5').className = min===5 ? "flex-1 p-2 rounded active-pomo" : "flex-1 p-2 rounded inactive-pomo";
    }
    function startPomo() {
      clearInterval(pomoInterval);
      pomoInterval = setInterval(()=>{
        pomoSec--;
        const m = String(Math.floor(pomoSec/60)).padStart(2,'0');
        const s = String(pomoSec%60).padStart(2,'0');
        document.getElementById('pomodisplay').innerText = m+":"+s;
        if(pomoSec <= 0) {
          clearInterval(pomoInterval);
          new Audio('https://assets.mixkit.co/sfx/preview/mixkit-software-interface-start-2574.mp3').play();
          alert("時間到！");
        }
      },1000);
    }
    function stopPomo() { clearInterval(pomoInterval); }
    function resetPomo() { pomoSec=25*60; setPomo(25); }

    function convertUnit() {
      const v = parseFloat(document.getElementById('val1').value)||1;
      const u1 = document.getElementById('unit1').value;
      const u2 = document.getElementById('unit2').value;
      let result = 0;
      let text = "";
      const map = {
        kg_lb:2.2, lb_kg:1/2.2,
        g_oz:0.035, oz_g:28.35,
        m_ft:3.28, ft_m:1/3.28,
        cm_inch:0.39, inch_cm:2.54
      };
      result = v * (map[u1+'_'+u2]||1);
      document.getElementById('val2').value = result.toFixed(2);
      document.getElementById('unitinfo').innerText = `1 ${u1} = ${(map[u1+'_'+u2]||1).toFixed(2)} ${u2}`;
    }

    function convertRate() {
      const v = document.getElementById('amt1').value||1;
      const c1 = document.getElementById('cur1').value;
      const c2 = document.getElementById('cur2').value;
      const rate = {HKD:1,USD:7.8,CNY:1.1,JPY:0.053,KRW:0.0058,SGD:5.8,THB:0.22,EUR:8.4,GBP:9.8,TWD:0.25};
      const res = v * rate[c1] / rate[c2];
      document.getElementById('amt2').value = res.toFixed(2);
      document.getElementById('rateinfo').innerText = `1 ${c1} = ${(rate[c1]/rate[c2]).toFixed(4)} ${c2}`;
    }

    function loadTodoMonths() {
      const p = document.getElementById('todoPage1');
      p.innerHTML = '';
      for(let m=1;m<=12;m++){
        p.innerHTML += `<div onclick="openMonth(${m})" class="bg-white p-3 rounded text-center">${m}月</div>`;
      }
    }
    function openMonth(m) {
      document.getElementById('todoPage1').classList.add('hidden');
      const p = document.getElementById('todoPage2');
      p.classList.remove('hidden');
      p.innerHTML = `<div onclick="backMonth1()" class="font-bold mb-2">← 返回</div><h3>2026年${m}月</h3>`;
      for(let d=1;d<=30;d++){
        p.innerHTML += `<div onclick="openDay(${m},${d})" class="inline-block w-10 h-10 m-1 bg-white rounded text-center leading-10">${d}</div>`;
      }
    }
    function backMonth1() {
      document.getElementById('todoPage2').classList.add('hidden');
      document.getElementById('todoPage1').classList.remove('hidden');
    }
    function openDay(m,d) {
      document.getElementById('todoPage2').classList.add('hidden');
      const p = document.getElementById('todoPage3');
      p.classList.remove('hidden');
      p.innerHTML = `
      <div onclick="backMonth2()" class="font-bold mb-2">← 返回</div>
      <h3>2026年${m}月${d}日</h3>
      <input id="todoInput" class="w-full p-2 border rounded my-2" placeholder="輸入待辦">
      <button onclick="addTodo()" class="bg-primary text-white p-2 rounded w-full mb-3">+ 新增</button>
      <div id="todoList"></div>`;
      loadTodoList(m,d);
    }
    function backMonth2() {
      document.getElementById('todoPage3').classList.add('hidden');
      document.getElementById('todoPage2').classList.remove('hidden');
    }
    function addTodo() {
      const m = document.querySelector('h3').innerText.match(/(\d+)年(\d+)月(\d+)日/)[2];
      const d = document.querySelector('h3').innerText.match(/(\d+)年(\d+)月(\d+)日/)[3];
      const txt = document.getElementById('todoInput').value;
      if(!txt) return;
      const key = `todo_2026${m}${d}`;
      const list = JSON.parse(localStorage.getItem(key)||'[]');
      list.push({txt:txt,done:false});
      localStorage.setItem(key,JSON.stringify(list));
      loadTodoList(m,d);
      document.getElementById('todoInput').value='';
    }
    function loadTodoList(m,d) {
      const key = `todo_2026${m}${d}`;
      const list = JSON.parse(localStorage.getItem(key)||'[]');
      const out = document.getElementById('todoList');
      out.innerHTML = '';
      list.forEach((it,i)=>{
        out.innerHTML += `<div class="p-2 bg-white rounded my-1 flex items-center gap-2" onclick="toggleTodo(${m},${d},${i})">
        <span class="w-5 h-5 border rounded">${it.done?'✅':''}</span>${it.txt}</div>`;
      });
    }
    function toggleTodo(m,d,i) {
      const key = `todo_2026${m}${d}`;
      const list = JSON.parse(localStorage.getItem(key)||'[]');
      list[i].done = !list[i].done;
      localStorage.setItem(key,JSON.stringify(list));
      loadTodoList(m,d);
    }
